<script>


const STATE = {

  IDLE: "idle",

  SPIN: "spin",

  RESULT: "result"

};


let state = STATE.IDLE;


let locked = false;


const items = [

  "Салфетки",

  "Кофе",

  "1л бензина",

  "2л бензина",

  "100₽",

  "Сладкий бонус",

  "День X"

];


const caseBtn = document.getElementById("case");

const roulette = document.getElementById("roulette");

const strip = document.getElementById("strip");

const result = document.getElementById("result");


function setState(s){

  state = s;

}


function buildStrip(){

  strip.innerHTML = "";

  for(let i=0;i<40;i++){

    const div = document.createElement("div");

    div.className = "item";

    div.textContent = items[Math.floor(Math.random()*items.length)];

    strip.appendChild(div);

  }

}


function spin(){

  if(locked) return;

  locked = true;


  setState(STATE.SPIN);


  buildStrip();


  roulette.style.display = "block";


  setTimeout(()=>{

    const offset = -(120*25);

    strip.style.transform = `translateX(${offset}px)`;


    setTimeout(()=>{

      showResult();

    }, 6000);


  },100);

}


function showResult(){

  setState(STATE.RESULT);


  const win = items[Math.floor(Math.random()*items.length)];


  result.style.display = "block";

  result.textContent = "Выигрыш: " + win;


  locked = false;

}


function reset(){

  setState(STATE.IDLE);

  roulette.style.display = "none";

  result.style.display = "none";

  strip.style.transform = "translateX(0)";

}


caseBtn.onclick = spin;


document.addEventListener("keydown",(e)=>{

  if(e.key === "Enter"){

    if(state === STATE.IDLE) spin();

    else if(state === STATE.RESULT) reset();

  }

});


document.getElementById("adminBtn").onclick = ()=>{

  document.getElementById("panel").classList.toggle("open");

};


</script>
